Table cliente {
  id integer [primary key]
  nombre varchar
  direccion varchar
  correo_electronico varchar
}

Table producto {
  id integer [primary key]
  nombre varchar
  precio real
  descripcion varchar
  categoria_id integer [ref: < categoria.id]
  proveedor_id integer [ref: < proveedor.id]
}

Table orden {
  id integer [primary key]
  fecha_de_orden date
  cliente_id integer [ref: < cliente.id]
}

Table categoria {
  id integer [primary key]
  nombre varchar
}

Table proveedor {
  id integer [primary key]
  nombre varchar
  direccion varchar
  correo_electronico varchar
}
